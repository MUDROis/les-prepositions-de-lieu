```js
function renderQuestions() {
  const container = document.getElementById('questions-container');
  questions.forEach((q, idx) => {
    const card = document.createElement('div');
    card.className = 'question-card';
    card.id = `q-${idx}`;

    const scene = document.createElement('div');
    scene.className = 'question-scene';
    scene.textContent = q.scene;

    const qText = document.createElement('div');
    qText.className = 'question-text';
    qText.innerHTML = `Où est le chat? <button class="speak-btn" data-text="Où est le chat" onclick="speak('Où est le chat')">🔊</button>`;

    const label = document.createElement('p');
    label.style.cssText = 'font-size:0.85rem;color:#888;text-align:center;margin-bottom:8px;';
    label.textContent = q.label;

    const optionsDiv = document.createElement('div');
    optionsDiv.className = 'options';

    q.options.forEach((opt, optIdx) => {
      const btn = document.createElement('button');
      btn.className = 'option-btn';
      btn.innerHTML = `${String.fromCharCode(97 + optIdx)}. ${opt.text} <button class="speak-btn" data-text="${opt.text.replace(/<[^>]*>/g, '').trim()}" onclick="event.stopPropagation(); speak('${opt.text.replace(/<[^>]*>/g, '').trim()}')">🔊</button>`;
      btn.onclick = () => handleAnswer(idx, optIdx);
      btn.id = `q-${idx}-opt-${optIdx}`;
      optionsDiv.appendChild(btn);
    });

    const explanation = document.createElement('div');
    explanation.className = 'explanation';
    explanation.id = `q-${idx}-expl`;

    card.appendChild(scene);
    card.appendChild(qText);
    card.appendChild(label);
    card.appendChild(optionsDiv);
    card.appendChild(explanation);
    container.appendChild(card);
  });
}
```

- [ ] **Step 3: Add answer handler**

```js
function handleAnswer(qIdx, optIdx) {
  const q = questions[qIdx];
  const selectedBtn = document.getElementById(`q-${qIdx}-opt-${optIdx}`);
  const correctBtn = document.getElementById(`q-${qIdx}-opt-${q.correctIndex}`);
  const explDiv = document.getElementById(`q-${qIdx}-expl`);

  if (selectedBtn.disabled) return;

  const allOptions = document.querySelectorAll(`#q-${qIdx} .option-btn`);
  allOptions.forEach(b => b.disabled = true);

  if (optIdx === q.correctIndex) {
    selectedBtn.classList.add('correct');
    correctCount++;
  } else {
    selectedBtn.classList.add('wrong');
    correctBtn.classList.add('correct');
  }

  explDiv.textContent = q.options[optIdx].explanation;
  explDiv.classList.add('show');

  answered++;
  document.getElementById('progress-text').textContent = `Répondu: ${answered}/${totalQuestions}`;

  if (answered === totalQuestions) {
    showSummary();
  }
}
```

```js
function showSummary() {
  const section = document.getElementById('summary-section');
  section.style.display = 'block';

  const table = document.getElementById('summary-table');
  questions.forEach((q, idx) => {
    const row = table.insertRow();
    const cellNum = row.insertCell();
    const cellAnswer = row.insertCell();
    const cellPrep = row.insertCell();
    const cellMeaning = row.insertCell();
    const cellCheck = row.insertCell();

    const chosenOpt = document.querySelector(`#q-${idx} .option-btn.correct, #q-${idx} .option-btn.wrong`);
    const chosenIdx = Array.from(document.querySelectorAll(`#q-${idx} .option-btn`)).indexOf(chosenOpt);

    cellNum.textContent = idx + 1;
    // Extract preposition from the chosen answer text
    const answerText = q.options[chosenIdx].text;
    const prepMatch = answerText.match(/\b(SUR|SOUS|DANS)\b/);
    const chosenPrep = prepMatch ? prepMatch[1] : '—';

    cellAnswer.textContent = String.fromCharCode(97 + chosenIdx);
    cellPrep.textContent = chosenPrep;
    cellMeaning.textContent = chosenPrep === 'SUR' ? 'on' : chosenPrep === 'SOUS' ? 'under' : chosenPrep === 'DANS' ? 'in' : '—';
    cellCheck.textContent = chosenIdx === q.correctIndex ? '✅' : '❌';
  });
}
```

- [ ] **Step 5: Add speech function and init**

```js
function speak(text) {
  if (!window.speechSynthesis) return;
  window.speechSynthesis.cancel();
  const utterance = new SpeechSynthesisUtterance(text);
  utterance.lang = 'fr-FR';
  utterance.rate = 0.85;

  const voices = window.speechSynthesis.getVoices();
  const frenchVoice = voices.find(v => v.lang.startsWith('fr'));
  if (frenchVoice) utterance.voice = frenchVoice;

  window.speechSynthesis.speak(utterance);
}

document.addEventListener('DOMContentLoaded', () => {
  renderQuestions();
  if (window.speechSynthesis) {
    window.speechSynthesis.getVoices();
  }
});
```
